🗄 SQL для Supabase

Выполните в SQL Editor вашего проекта:

CREATE TABLE IF NOT EXISTS land_reports (
  id               BIGSERIAL PRIMARY KEY,
  report_date      DATE          NOT NULL,
  report_name      TEXT,
  plots_count      INTEGER       NOT NULL DEFAULT 0,
  seizure_area     NUMERIC(14,4) NOT NULL DEFAULT 0,
  cadastral_amount NUMERIC(20,2) NOT NULL DEFAULT 0,
  created_at       TIMESTAMPTZ   NOT NULL DEFAULT NOW()
);
CREATE INDEX IF NOT EXISTS idx_land_reports_date    ON land_reports(report_date);
CREATE INDEX IF NOT EXISTS idx_land_reports_created ON land_reports(created_at);
ALTER TABLE land_reports ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Allow anon read"   ON land_reports FOR SELECT USING (true);
CREATE POLICY "Allow anon insert" ON land_reports FOR INSERT WITH CHECK (true);
CREATE POLICY "Allow anon delete" ON land_reports FOR DELETE USING (true);
-- ⚠ ВАЖНО: Если таблица уже существует, добавьте колонку одной командой:
ALTER TABLE land_reports ADD COLUMN IF NOT EXISTS report_name TEXT;
⚠️

Удалить запись?

Это действие необратимо. Запись будет удалена из Supabase.
